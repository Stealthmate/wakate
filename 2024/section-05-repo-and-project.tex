\section{To モノレポ or not to モノレポ?}

\begin{frame}
    \frametitle{どういうときにモノレポが良いのか？}
    \small
    \begin{block}{大前提}
        \begin{enumerate}
            \item<2-> 関係あるもの同士を同じ個体にまとめるのは管理コストを下げる．
            \item<3-> 個体を分割することは管理コストを上げる．
        \end{enumerate}
    \end{block}
    \onslide<4>{
        \Large\emphasize{=> 原則モノレポが良い．}
    }
\end{frame}
