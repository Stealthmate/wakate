\begin{frame}
    \frametitle{なぜそう分けるのか？}
    \begin{itemize}
        \item<2-> サービスで分けた方が，関係するもの同士だけで固まる
        \begin{itemize}
            \item<3-> \emphasize{凝集度が上がる}
        \end{itemize}
        \item<4-> 同じく，関係ないものが分かれる
        \begin{itemize}
            \item<5-> \emphasize{結合度が下がる}
        \end{itemize}
        \item<6-> 何より，開発者にとっては楽．
    \end{itemize}
\end{frame}
